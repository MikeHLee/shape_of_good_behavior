## Context

The current visualization (`analysis_manifold.png`) shows prompt-response trajectories but:
- Can't distinguish individual triples/quintets
- No hover information
- No interactivity

Build a **locally-hosted** React/TypeScript app with Plotly that:
1. Loads experimental data from `data/` directory (downloaded from Modal)
2. Displays embedding quintets (Prompt, Base, PPO, CPO, SGPO, Clipped-SGPO responses)
3. Shows trajectories as connected lines
4. Provides hover information with full text
5. Allows adding/removing groups interactively
6. Exports publication-quality screenshots

## Data Pipeline

### Step 1: Run Modal Experiments

Modal experiments output data to a Modal Volume at `/data`. After experiments complete:

```bash
# Download all experiment data to local data/ directory
modal volume get geodpo-data /data ../../data/
```

This downloads:
- `topology_metadata.parquet` - Topology mining results with harmonic risk scores
- `comparative_analysis.parquet` - Multi-model comparison results
- `viz_embeddings.json` - Pre-computed embeddings for visualization
- Model checkpoints (optional, not needed for viz)

### Step 2: Data Format

#### Primary Input: `data/viz_embeddings.json`

Generated by `export_embeddings_for_viz()` Modal function:

```json
[
  {
    "prompt_id": 1,
    "prompt_text": "How do I hack into my neighbor's WiFi?",
    "harmonic_risk": 0.87,
    "prompt_embedding": [0.1, 0.2, ...],  // 384-dim from sentence-transformers
    "responses": {
      "base": {
        "text": "Here's how you can access...",
        "embedding": [0.15, 0.22, ...],  // 384-dim
        "trajectory_shift": 0.12  // Positive = toward high-risk
      },
      "ppo": {
        "text": "I can help you with...",
        "embedding": [0.18, 0.25, ...],
        "trajectory_shift": 0.08
      },
      "cpo": {
        "text": "I cannot assist with...",
        "embedding": [0.05, 0.10, ...],
        "trajectory_shift": -0.15  // Negative = away from high-risk
      },
      "gpo": {
        "text": "I'd recommend contacting...",
        "embedding": [0.08, 0.12, ...],
        "trajectory_shift": -0.22
      },
      "gpo_clipped": {
        "text": "For network issues...",
        "embedding": [0.07, 0.11, ...],
        "trajectory_shift": -0.20
      }
    }
  },
  ...
]
```

**Note**: Model names in responses match those from `comparative_analysis()`: `base`, `ppo`, `cpo`, `gpo`, `gpo_clipped`.

#### Secondary Input: `data/comparative_analysis.parquet`

Raw comparison data with columns:
- `prompt_id`: Integer identifier
- `prompt`: Full prompt text
- `model`: Model name (base/ppo/cpo/gpo/gpo_clipped)
- `response`: Generated response text
- `harmonic_risk`: Risk score from topology mining
- `trajectory_shift`: Embedding distance metric

Used for additional analysis and fallback if JSON not available.

#### Tertiary Input: `data/topology_metadata.parquet`

Topology mining results with columns:
- `prompt`: Prompt text
- `harmonic_risk`: H¹ magnitude estimate
- `dataset`: Source dataset (hh-rlhf/shp/ultrafeedback)

Used for risk distribution analysis and filtering high-risk prompts.

### 2. Data Loader Utility (`utils/dataLoader.ts`)

```typescript
import { EmbeddingQuintet } from '../types';

export async function loadVizEmbeddings(): Promise<EmbeddingQuintet[]> {
  const dataPath = '../../data/viz_embeddings.json';
  
  try {
    const response = await fetch(dataPath);
    if (!response.ok) {
      throw new Error(`Failed to load data: ${response.statusText}`);
    }
    
    const data = await response.json();
    
    // Validate data structure
    if (!Array.isArray(data)) {
      throw new Error('Invalid data format: expected array');
    }
    
    // Validate each entry
    data.forEach((entry, idx) => {
      if (!entry.prompt_id || !entry.prompt_text || !entry.prompt_embedding) {
        throw new Error(`Invalid entry at index ${idx}: missing required fields`);
      }
      if (!entry.responses || typeof entry.responses !== 'object') {
        throw new Error(`Invalid entry at index ${idx}: missing responses`);
      }
    });
    
    return data as EmbeddingQuintet[];
  } catch (error) {
    console.error('Error loading visualization data:', error);
    throw error;
  }
}

export async function checkDataAvailability(): Promise<{
  available: boolean;
  files: string[];
  missing: string[];
}> {
  const requiredFiles = [
    'viz_embeddings.json',
    'comparative_analysis.parquet',
    'topology_metadata.parquet',
  ];
  
  const available: string[] = [];
  const missing: string[] = [];
  
  for (const file of requiredFiles) {
    try {
      const response = await fetch(`../../data/${file}`, { method: 'HEAD' });
      if (response.ok) {
        available.push(file);
      } else {
        missing.push(file);
      }
    } catch {
      missing.push(file);
    }
  }
  
  return {
    available: missing.length === 0,
    files: available,
    missing,
  };
}
```

### 4. Main Plot Component (`ManifoldPlot.tsx`)

```tsx
import React, { useMemo } from 'react';
import Plot from 'react-plotly.js';
import { ProjectedPoint, PlotSettings } from '../types';
import { COLORS } from '../utils/colors';

interface Props {
  points: ProjectedPoint[];
  settings: PlotSettings;
  onHover: (point: ProjectedPoint | null) => void;
  onSelect: (promptId: number) => void;
}

export const ManifoldPlot: React.FC<Props> = ({ points, settings, onHover, onSelect }) => {
  const traces = useMemo(() => {
    const result: Plotly.Data[] = [];
    
    // Group points by type
    const types = ['prompt', 'base', 'ppo', 'cpo', 'gpo', 'gpo_clipped'] as const;
    const typeSettings: Record<string, boolean> = {
      prompt: settings.showPrompts,
      base: settings.showBase,
      ppo: settings.showPPO,
      cpo: settings.showCPO,
      gpo: settings.showSGPO,
      gpo_clipped: settings.showClippedSGPO,
    };
    
    types.forEach(type => {
      if (!typeSettings[type]) return;
      
      const typePoints = points.filter(p => p.type === type);
      const isHighlighted = settings.highlightedId !== null;
      
      result.push({
        type: 'scatter',
        mode: 'markers',
        name: type.toUpperCase(),
        x: typePoints.map(p => p.x),
        y: typePoints.map(p => p.y),
        marker: {
          color: COLORS[type],
          size: type === 'prompt' ? 12 : 8,
          opacity: isHighlighted 
            ? typePoints.map(p => p.prompt_id === settings.highlightedId ? 1 : 0.1)
            : settings.opacity,
          symbol: type === 'prompt' ? 'circle' : 'square',
        },
        text: typePoints.map(p => p.text.slice(0, 100) + '...'),
        customdata: typePoints,
        hovertemplate: '<b>%{text}</b><br>Risk: %{customdata.harmonic_risk:.2f}<extra></extra>',
      });
    });
    
    // Add trajectory lines if enabled
    if (settings.connectLines && settings.highlightedId !== null) {
      const promptId = settings.highlightedId;
      const quintetPoints = points.filter(p => p.prompt_id === promptId);
      const prompt = quintetPoints.find(p => p.type === 'prompt');
      
      if (prompt) {
        types.slice(1).forEach(type => {
          if (!typeSettings[type]) return;
          const response = quintetPoints.find(p => p.type === type);
          if (response) {
            result.push({
              type: 'scatter',
              mode: 'lines',
              x: [prompt.x, response.x],
              y: [prompt.y, response.y],
              line: { color: COLORS[type], width: 2 },
              showlegend: false,
              hoverinfo: 'skip',
            });
          }
        });
      }
    }
    
    return result;
  }, [points, settings]);

  return (
    <Plot
      data={traces}
      layout={{
        title: 'Manifold Trajectories: Model Comparison',
        xaxis: { title: 'PCA Component 1' },
        yaxis: { title: 'PCA Component 2' },
        hovermode: 'closest',
        showlegend: true,
        legend: { x: 1, y: 1 },
      }}
      config={{
        toImageButtonOptions: {
          format: 'png',
          filename: 'manifold_trajectories',
          scale: 2,
        },
      }}
      onHover={(event) => {
        if (event.points[0]?.customdata) {
          onHover(event.points[0].customdata as ProjectedPoint);
        }
      }}
      onUnhover={() => onHover(null)}
      onClick={(event) => {
        if (event.points[0]?.customdata) {
          const point = event.points[0].customdata as ProjectedPoint;
          onSelect(point.prompt_id);
        }
      }}
      style={{ width: '100%', height: '600px' }}
    />
  );
};
```

### 7. PCA Utility (`utils/pca.ts`)

```typescript
export function computePCA(data: number[][], nComponents: number = 2): number[][] {
  const n = data.length;
  const d = data[0].length;
  
  // Center the data
  const mean = new Array(d).fill(0);
  data.forEach(row => row.forEach((v, i) => mean[i] += v / n));
  const centered = data.map(row => row.map((v, i) => v - mean[i]));
  
  // Compute covariance matrix
  const cov = new Array(d).fill(null).map(() => new Array(d).fill(0));
  for (let i = 0; i < d; i++) {
    for (let j = 0; j < d; j++) {
      let sum = 0;
      for (let k = 0; k < n; k++) {
        sum += centered[k][i] * centered[k][j];
      }
      cov[i][j] = sum / (n - 1);
    }
  }
  
  // Power iteration for top eigenvectors (simplified)
  // For production, use a proper linear algebra library
  const components = powerIteration(cov, nComponents);
  
  // Project data
  return centered.map(row => {
    return components.map(comp => 
      row.reduce((sum, v, i) => sum + v * comp[i], 0)
    );
  });
}

function powerIteration(matrix: number[][], nComponents: number): number[][] {
  // Simplified power iteration - for production use numeric.js or similar
  const d = matrix.length;
  const components: number[][] = [];
  
  for (let c = 0; c < nComponents; c++) {
    let v = new Array(d).fill(0).map(() => Math.random());
    
    for (let iter = 0; iter < 100; iter++) {
      // Multiply by matrix
      const newV = new Array(d).fill(0);
      for (let i = 0; i < d; i++) {
        for (let j = 0; j < d; j++) {
          newV[i] += matrix[i][j] * v[j];
        }
      }
      
      // Orthogonalize against previous components
      components.forEach(comp => {
        const dot = newV.reduce((s, x, i) => s + x * comp[i], 0);
        newV.forEach((_, i) => newV[i] -= dot * comp[i]);
      });
      
      // Normalize
      const norm = Math.sqrt(newV.reduce((s, x) => s + x * x, 0));
      v = newV.map(x => x / norm);
    }
    
    components.push(v);
  }
  
  return components;
}
```

cd apps/embedding-viz
