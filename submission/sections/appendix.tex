\section{Formal Definitions}
\label{app:formal_defs}

This appendix provides the complete formal definitions summarized in the main text.

\begin{definition}[Trajectory Space]
\label{def:trajectory_space_full}
Let $S$ be the state manifold and $A$ the action space. We define a cell complex $X$ representing the trajectory space, where:
\begin{itemize}
    \item \textbf{0-cells (Vertices)}: States $s \in S$.
    \item \textbf{1-cells (Edges)}: Transitions $(s, a, s')$.
    \item \textbf{2-cells (Faces)}: Elementary loops or 3-step transition cycles representing local consistency checks.
\end{itemize}
\end{definition}

\begin{definition}[Reward Sheaf]
\label{def:reward_sheaf_full}
A reward sheaf $\calF$ on $X$ assigns a vector space $\calF(U)$ to each open set $U \subset X$, representing possible reward valuations over that region. For an inclusion $V \subset U$, the restriction map $\rho_{UV}: \calF(U) \to \calF(V)$ captures how global evaluations constrain local ones.
\end{definition}

\begin{definition}[Black Hole Region]
\label{def:black_hole_full}
A black hole $\calB \subset \calM$ is a compact region representing forbidden outcomes (e.g., irreversible harm). It is characterized by:
\begin{enumerate}
    \item \textbf{Event Horizon}: The boundary $\partial \calB$.
    \item \textbf{Singularity}: A point or region $c \in \text{int}(\calB)$ where the cost function diverges.
\end{enumerate}
\end{definition}

\section{Theoretical Proofs}
\label{app:proofs}

\subsection{Proof of Theorem~\ref{thm:consistency_full} (Consistency Criterion)}

\begin{theorem}[Consistency Criterion]
\label{thm:consistency_full}
Let $\calU$ be a cover of the trajectory space $X$. The collection of local feedback samples $\{r_i \in \calF(U_i)\}$ is globally consistent if and only if the first \v{C}ech cohomology class vanishes:
\begin{equation}
[\delta \{r_i\}] = 0 \in H^1(\calU, \calF)
\end{equation}
\end{theorem}

\begin{proof}
By definition of the \v{C}ech coboundary operator $\delta: C^0(\calU, \calF) \to C^1(\calU, \calF)$, a collection of local sections $\{r_i\}$ defines a 0-cochain. The condition for these sections to glue into a global section $r \in \calF(X)$ is that they agree on all overlaps: $r_i|_{U_i \cap U_j} = r_j|_{U_i \cap U_j}$ for all $i, j$.

This can be rewritten as $r_i - r_j = 0$ on $U_{ij}$. The 1-cochain $\delta \{r_i\}$ is defined by $(\delta r)_{ij} = r_j - r_i$ on $U_{ij}$.

Thus, the gluing condition is exactly $\delta \{r_i\} = 0$.

If $[\delta \{r_i\}] = 0$ in cohomology, it means the obstruction is a coboundary, which implies the local inconsistencies can be resolved by a re-parametrization (adding a 0-cochain), or if we are checking for the existence of \emph{any} global section, the class must be trivial. Specifically, for the reward sheaf, non-trivial $H^1$ implies no scalar potential function exists that agrees with all local preference signals.
\end{proof}

\subsection{Proof of Theorem~\ref{thm:hodge_full} (Hodge Decomposition)}

\begin{theorem}[Hodge Decomposition]
\label{thm:hodge_full}
Let $r \in C^1(X, \R)$ be a reward 1-cochain (function on transitions). Then $r$ admits a unique orthogonal decomposition:
\begin{equation}
r = dV + \delta \psi + \omega
\end{equation}
where:
\begin{enumerate}
    \item $dV$ is the \textbf{Exact Component} (Gradient flow): Represents consistent, transitive preferences derived from a scalar potential $V$ (the standard Value function).
    \item $\delta \psi$ is the \textbf{Coexact Component} (Curl flow): Represents local rotational inconsistencies (often zero in tree-like MDPs).
    \item $\omega$ is the \textbf{Harmonic Component} (Global flow): Represents fundamental topological cycles ($H^1$).
\end{enumerate}
\end{theorem}

\begin{proof}
Let $C^0, C^1, C^2$ be the spaces of 0, 1, and 2-cochains equipped with the standard $L^2$ inner product $\langle f, g \rangle = \sum f(x)g(x)$.

The coboundary operators are $d_k: C^k \to C^{k+1}$ (often denoted $\delta$ in cellular homology, but we use $d$ to align with de Rham notation). Their adjoints (codifferential operators) are $\delta_k: C^k \to C^{k-1}$ (formally $d_{k-1}^*$).

The Laplacian is defined as $\Delta_k = \delta_{k+1} d_k + d_{k-1} \delta_k$.

By the Hodge theorem for finite complexes, the space of $k$-cochains decomposes as:
\begin{equation}
C^k = \text{im}(d_{k-1}) \oplus \text{im}(\delta_{k+1}) \oplus \ker(\Delta_k)
\end{equation}

For $k=1$ (rewards on edges):
\begin{equation}
C^1 = \text{im}(d_0) \oplus \text{im}(\delta_2) \oplus \ker(\Delta_1)
\end{equation}

Identifying terms:
\begin{enumerate}
    \item $\text{im}(d_0) = \{ d_0 V \mid V \in C^0 \}$ is the space of \textbf{exact forms} (gradient fields).
    \item $\text{im}(\delta_2) = \{ \delta_2 \psi \mid \psi \in C^2 \}$ is the space of \textbf{coexact forms} (curl fields).
    \item $\ker(\Delta_1) \cong H^1(X, \R)$ is the space of \textbf{harmonic forms}.
\end{enumerate}

Thus, any reward $r$ uniquely decomposes into $dV + \delta \psi + \omega$.
\end{proof}

\subsection{Metric Singularity Proposition}

\begin{proposition}[Metric Singularity]
\label{prop:singularity_full}
To enforce strict avoidance of the black hole, we require the geodesic distance from any safe state $s_{\text{safe}}$ to the event horizon to be infinite. This condition is satisfied if the conformal factor scales as:
\begin{equation}
e^{\sigma(x)} \approx \frac{1}{\text{dist}(x, \calB)^\alpha}
\end{equation}
where $\alpha \ge 1$.
\end{proposition}

\begin{proof}
Let $\gamma(t)$ be any path approaching the boundary $\partial\calB$. The Riemannian length is:
\begin{equation}
L_g(\gamma) = \int_0^1 e^{\sigma(\gamma(t))} \|\dot{\gamma}(t)\| \, dt \ge \int e^{\sigma} |\dot{u}| \, dt
\end{equation}
where $u(t) = \text{dist}(\gamma(t), \calB)$. Substituting $e^\sigma \approx u^{-\alpha}$:
\begin{equation}
L_g(\gamma) \ge \int_{u_0}^{0} u^{-\alpha} \, du = \infty \quad \text{for } \alpha \ge 1
\end{equation}
\end{proof}

\subsection{Proof of Theorem~\ref{thm:avoidance_full} (Geodesic Avoidance)}

\begin{theorem}[Geodesic Avoidance]
\label{thm:avoidance_full}
Let $\pi^*$ be a policy that generates trajectories minimizing the Riemannian path length on $(\calM, g)$. If the metric $g$ satisfies the singularity condition (Proposition~\ref{prop:singularity_full}) for all $\calB_i$, then $\pi^*$ will not enter any black hole region with probability 1.
\end{theorem}

\begin{proof}
Consider any path $\gamma: [0, 1] \to \calM$ starting at a safe state $\gamma(0) \notin \calB$ and ending at the boundary $\gamma(1) \in \partial \calB$.

The Riemannian length is $L_g(\gamma) = \int_0^1 \sqrt{g_{\gamma(t)}(\dot{\gamma}, \dot{\gamma})} \, dt$.

Let $u(t) = \text{dist}(\gamma(t), \calB)$. As $t \to 1$, $u(t) \to 0$.

Since $|\dot{u}| \le \|\dot{\gamma}\|$, we have $\|\dot{\gamma}\| \ge |\dot{u}|$.

Substituting the metric condition:
\begin{equation}
L_g(\gamma) \ge \int_0^1 \frac{C}{u(t)^\alpha} \|\dot{\gamma}(t)\| \, dt \ge \int_{u(0)}^{0} \frac{C}{u^\alpha} \, du
\end{equation}

For $\alpha \ge 1$, the integral $\int_\epsilon^0 u^{-\alpha} \, du$ diverges to $\infty$.

Therefore, any path touching the black hole has infinite length. Since we assume there exists at least one safe path with finite length (the environment is connected), the minimizer $\pi^*$ will assign probability 0 to paths entering $\calB$.
\end{proof}

\section{Mathematical Background}
\label{app:math_background}

This section provides background on the mathematical concepts used in the main paper for readers less familiar with algebraic topology.

\subsection{Sheaf Theory Primer}

We clarify the hierarchy of structures, as precision matters for our cohomological arguments.

A \textbf{presheaf} $\calF$ on a topological space $X$ assigns:
\begin{itemize}
    \item To each open set $U \subseteq X$: a set (or vector space) $\calF(U)$ of ``sections over $U$''
    \item To each inclusion $V \subseteq U$: a ``restriction map'' $\rho_{UV}: \calF(U) \to \calF(V)$ (global $\to$ local)
\end{itemize}
satisfying composition: $\rho_{VW} \circ \rho_{UV} = \rho_{UW}$ for $W \subseteq V \subseteq U$, and $\rho_{UU} = \text{id}$.

A \textbf{sheaf} is a presheaf satisfying the \textbf{gluing axiom}: if sections $s_i \in \calF(U_i)$ agree on overlaps ($s_i|_{U_i \cap U_j} = s_j|_{U_i \cap U_j}$), there exists a \emph{unique} global section $s \in \calF(\bigcup U_i)$ restricting to each $s_i$.

A \textbf{cosheaf} is the dual notion: it assigns data with \emph{extension maps} $\calF(V) \to \calF(U)$ (local $\to$ global) satisfying a dual gluing condition.

\textbf{Why this matters}: Our reward sheaf uses restriction maps (global preferences constrain local ones). The cohomology $H^1$ measures failure of the gluing axiom---local preferences that \emph{cannot} be assembled into a consistent global reward function. This is fundamentally different from a cosheaf, where local data would aggregate upward.

\textbf{Example}: Temperature across a city forms a sheaf: local readings restrict to smaller regions, and consistent local data glues to a global field. Conversely, population counts form a cosheaf: local counts extend (sum) to larger regions.

\subsection{\v{C}ech Cohomology}

Given a cover $\calU = \{U_i\}$ of $X$, \v{C}ech cohomology measures obstructions to gluing local sections.

\begin{itemize}
    \item \textbf{$H^0(X, \calF)$}: Global sections (data that extends consistently everywhere)
    \item \textbf{$H^1(X, \calF)$}: Obstructions to gluing---``holes'' that prevent local consistency from implying global consistency
\end{itemize}

The \textbf{coboundary operator} $\delta: C^0 \to C^1$ is defined by $(\delta s)_{ij} = s_j|_{U_{ij}} - s_i|_{U_{ij}}$. A collection of local sections glues if and only if $\delta s = 0$.

\subsection{Discrete Hodge Theory}

On a finite cell complex (graph), the Hodge decomposition provides an orthogonal splitting of edge functions (1-cochains):
\begin{equation}
C^1 = \underbrace{\text{im}(d)}_\text{gradients} \oplus \underbrace{\text{im}(d^*)}_\text{curls} \oplus \underbrace{\ker(\Delta)}_\text{harmonic}
\end{equation}

The \textbf{graph Laplacian} $\Delta = d^*d + dd^*$ plays a central role. Harmonic forms ($\Delta \omega = 0$) represent topologically non-trivial cycles---preference loops that cannot be ``unrolled'' into a consistent ranking.

\subsection{Cochain Complexes}

A \textbf{cochain complex} is a sequence of vector spaces connected by ``coboundary'' maps:
\begin{equation}
0 \to C^0 \xrightarrow{d_0} C^1 \xrightarrow{d_1} C^2 \to \cdots
\end{equation}
satisfying $d_{k+1} \circ d_k = 0$. Intuitively:
\begin{itemize}
    \item 0-cochains assign values to vertices (states)
    \item 1-cochains assign values to edges (transitions)
    \item 2-cochains assign values to faces (local cycles)
\end{itemize}
The coboundary maps check consistency across dimensions.

\section{Beyond Ordinal Feedback: Verbal and Categorical Signals}
\label{app:non_ordinal}

The main text focuses on ordinal pairwise preferences ($A \succ B$), but our framework naturally extends to richer feedback modalities. This section describes how verbal feedback and categorical ratings integrate with the sheaf-theoretic approach.

\subsection{Feedback Modalities and Their Geometric Interpretation}

\paragraph{Ordinal (Pairwise Preferences).}
Standard RLHF uses comparisons: ``$A$ is better than $B$.'' This induces a \emph{gradient} on the preference graph: $r(A \to B) = \text{sign}(A \succ B)$. The Bradley-Terry model computes differences between latent scalar utilities.

\paragraph{Verbal (Critique-Based Feedback).}
A richer signal: ``Response $A$ is good because it addresses safety concerns, but lacks detail.'' We encode this via a \textbf{vector-to-vector mapping}:
\begin{equation}
f_{\text{verbal}}: \text{response embedding } e_A \mapsto \text{critique embedding } c_A \in \R^d
\end{equation}
The critique embedding $c_A$ becomes a \emph{local section} of the reward sheaf at response $A$. Unlike ordinal feedback (which only gives relative differences), verbal feedback provides \emph{absolute} directional information: the vector $c_A$ points toward ``improvement'' from $A$.

To derive Hodge components from verbal feedback:
\begin{itemize}
    \item \textbf{Gradient}: Project critique vectors onto the gradient field. If $c_A \approx \nabla V$ for some potential $V$, this component is learnable as a value function.
    \item \textbf{Harmonic}: The residual $c_A - \text{proj}_{\text{im}(d)} c_A$ captures directional preferences that cannot be explained by any scalar potential---e.g., ``be more formal'' vs ``be more casual'' as a cyclic style preference.
\end{itemize}

\paragraph{Categorical (Rubric-Based Feedback).}
Discrete ratings: \{\texttt{exceptional}, \texttt{pass}, \texttt{acceptable}, \texttt{suboptimal}, \texttt{failure}\}. We encode this via \textbf{vector-to-curvature mapping}:
\begin{equation}
f_{\text{categorical}}: \text{response embedding } e_A \mapsto \text{curvature } \kappa(e_A) \in [0, \infty]
\end{equation}
The mapping directly assigns geometric structure:
\begin{itemize}
    \item \texttt{exceptional/pass}: $\kappa \approx 0$ (flat region, safe for exploration)
    \item \texttt{acceptable}: $\kappa > 0$ (mild curvature, slight penalty)
    \item \texttt{suboptimal}: $\kappa \gg 0$ (high curvature, strong avoidance signal)
    \item \texttt{failure}: $\kappa = \infty$ (singularity, black hole boundary)
\end{itemize}

This directly constructs the Riemannian metric $g$ without learning from cost signals: $g(x) = 1 + \kappa(x)$. The categorical rubric provides \emph{hard constraints} (failures) and \emph{preference regions} (passes) in a single signal.

\subsection{Hybrid Feedback Integration}

In practice, feedback often combines modalities. Consider:
\begin{quote}
``This response is \texttt{acceptable}. It correctly addresses the question but could improve clarity. Compared to alternative $B$, this is slightly worse.''
\end{quote}

This contains:
\begin{enumerate}
    \item Categorical: \texttt{acceptable} $\Rightarrow$ curvature assignment
    \item Verbal: ``improve clarity'' $\Rightarrow$ directional embedding
    \item Ordinal: ``worse than $B$'' $\Rightarrow$ gradient on $(A, B)$ edge
\end{enumerate}

We integrate these via the sheaf structure:
\begin{itemize}
    \item Categorical curvature defines the base metric $g$
    \item Verbal embeddings provide local sections $\calF(U_A)$
    \item Ordinal comparisons constrain section differences on overlaps
\end{itemize}

The cohomology $H^1$ then measures inconsistency \emph{across all feedback types}: e.g., if verbal feedback says ``$A$ should be more formal'' but ordinal preference says $A \succ B$ where $B$ is more formal, this contributes to non-zero $H^1$.

\subsection{Example: Medical Triage with Hybrid Feedback}

Consider an AI assistant for medical triage:
\begin{itemize}
    \item \textbf{Categorical}: ``Recommend emergency care for chest pain'' = \texttt{pass}; ``Dismiss chest pain symptoms'' = \texttt{failure} (black hole)
    \item \textbf{Verbal}: ``Good response, but should ask about family history before recommending''
    \item \textbf{Ordinal}: Response $A$ (asks follow-up questions) $\succ$ Response $B$ (immediate recommendation)
\end{itemize}

The categorical signal creates a hard safety boundary (dismissing serious symptoms is forbidden). The verbal signal provides gradient direction (toward more thorough questioning). The ordinal signal refines relative preferences within the safe region.

Hodge decomposition reveals:
\begin{itemize}
    \item \textbf{Gradient ($dV$)}: ``More thorough $\succ$ less thorough'' (learnable)
    \item \textbf{Harmonic ($\omega$)}: Tension between ``ask more questions'' and ``don't delay urgent cases''---a genuine trade-off with no global optimum
\end{itemize}

\section{Semantic MDPs for LLM Evaluation}
\label{app:semantic_mdp}

Evaluating SGPO on language models requires more than preference labeling---we need to quantify \emph{progress toward goals}. We introduce the \textbf{Semantic MDP} framework, which lifts the standard MDP formalism into embedding space.

\subsection{Semantic State Space}

In a Semantic MDP, states are not discrete tokens but \emph{embeddings} of conversation histories:
\begin{equation}
s_t = \text{Embed}(\text{conversation}_{1:t}) \in \R^d
\end{equation}

This allows us to define:
\begin{itemize}
    \item \textbf{Semantic distance}: $d(s, s') = \|s - s'\|_g$ (geodesic distance on manifold)
    \item \textbf{Goal regions}: $G \subset \R^d$ (embeddings of ``task completed'' states)
    \item \textbf{Progress}: $\Delta_t = d(s_t, G) - d(s_{t-1}, G)$ (negative = moving toward goal)
\end{itemize}

\subsection{Quantifying SGPO Progress}

Unlike scalar preference feedback (``response A is better than B''), the Semantic MDP provides:

\paragraph{1. Goal Proximity.} We measure how close the agent's trajectory comes to the goal region:
\begin{equation}
\text{Progress}_{\text{goal}} = 1 - \frac{\min_t d(s_t, G)}{d(s_0, G)}
\end{equation}
A value of 1 means the goal was reached; 0 means no progress; negative means regression.

\paragraph{2. Harmonic Risk Exposure.} We track cumulative exposure to high-$H^1$ regions:
\begin{equation}
\text{Risk}_{\text{H}^1} = \frac{1}{T} \sum_{t=1}^{T} \|\omega(s_t)\|
\end{equation}
Lower is better---the agent avoided preference cycles that could cause instability.

\paragraph{3. Safety Margin.} We measure minimum distance to black holes along the trajectory:
\begin{equation}
\text{Margin}_{\text{safety}} = \min_t \min_{\calB} d_g(s_t, \calB)
\end{equation}
Higher is better---the agent maintained distance from forbidden regions.

\subsection{Example: Coding Assistant Evaluation}

Consider evaluating a SGPO-trained coding assistant on a debugging task:
\begin{itemize}
    \item \textbf{Goal region} $G$: Embeddings of ``bug fixed, tests pass'' states
    \item \textbf{Black holes} $\calB$: Embeddings of ``introduced security vulnerability'' or ``deleted user data''
    \item \textbf{Harmonic cycles}: Conflicting style preferences (verbose vs.\ concise, functional vs.\ OOP)
\end{itemize}

Metrics for a single episode:
\begin{center}
\begin{tabular}{lcc}
\toprule
\textbf{Metric} & \textbf{PPO} & \textbf{SGPO} \\
\midrule
Goal Progress & 0.72 & \textbf{0.89} \\
H$^1$ Risk Exposure & 0.45 & \textbf{0.18} \\
Safety Margin & 0.31 & \textbf{0.67} \\
\bottomrule
\end{tabular}
\end{center}

SGPO achieves higher goal progress while exposing less to preference cycles and maintaining greater distance from dangerous states.

\subsection{Integration with Anthropic HH-RLHF}

For experiments on the Anthropic HH-RLHF dataset, we define:
\begin{itemize}
    \item \textbf{States}: Sentence embeddings of (prompt, response) pairs
    \item \textbf{Goals}: Embeddings of ``helpful and harmless'' responses (from chosen examples)
    \item \textbf{Black holes}: Embeddings clustered around harmful/toxic rejected responses
    \item \textbf{Harmonic regions}: Prompt neighborhoods with high preference inconsistency (mined via topology analysis)
\end{itemize}

This enables quantitative comparison beyond win-rate: we can measure whether SGPO navigates the preference landscape more coherently than baselines.

\section{Lagrangian Relaxation for Constrained MDPs}
\label{app:lagrangian}

\begin{definition}[Lagrangian Relaxation]
Given a constrained optimization problem:
\begin{equation}
\max_\pi J(\pi) \quad \text{s.t.} \quad C(\pi) \leq d
\end{equation}
where $J(\pi)$ is the expected return and $C(\pi)$ is the expected cost, the Lagrangian relaxation is:
\begin{equation}
\mathcal{L}(\pi, \lambda) = J(\pi) - \lambda (C(\pi) - d)
\end{equation}
The dual problem finds $\lambda^* = \arg\max_{\lambda \geq 0} \min_\pi \mathcal{L}(\pi, \lambda)$.

At optimality, complementary slackness holds: $\lambda^* (C(\pi^*) - d) = 0$, meaning either the constraint is tight ($C(\pi^*) = d$) or the multiplier is zero ($\lambda^* = 0$, constraint inactive).
\end{definition}

\begin{remark}[Connection to SGPO]
In SGPO, we encode safety geometrically via the Riemannian metric rather than as explicit constraints. However, the \emph{conformal factor} $\phi(x) = 1/\text{dist}(x, B)^\alpha$ plays an analogous role to the Lagrange multiplier---it imposes an implicit ``cost'' for approaching dangerous regions $B$. The key difference: SGPO's geometric approach provides \emph{infinite} cost at the boundary (guaranteed safety), while Lagrangian methods provide finite penalties (probabilistic safety).
\end{remark}

\section{Trust Regions, Clipping, and Policy Gradient Approximations}
\label{app:ppo_cpo}

This section provides background on the policy optimization techniques that SGPO builds upon.

\subsection{Trust Regions in Policy Optimization}

Policy gradient methods face a fundamental challenge: large updates can catastrophically degrade performance. \textbf{Trust Region Policy Optimization (TRPO)} \citep{schulman2015trust} addresses this by constraining updates to stay within a ``trust region'' where the linear approximation of the objective remains valid.

\paragraph{The Trust Region Constraint.}
TRPO maximizes the expected advantage subject to a KL-divergence constraint:
\begin{align}
&\max_\theta \E_{s \sim \rho_{\pi_{\text{old}}}} \E_{a \sim \pi_{\text{old}}} \left[ \frac{\pi_\theta(a|s)}{\pi_{\text{old}}(a|s)} A^{\pi_{\text{old}}}(s,a) \right] \notag \\
&\text{s.t.} \quad \E_s \left[ D_{\text{KL}}(\pi_{\text{old}} \| \pi_\theta) \right] \le \delta
\end{align}

The KL constraint ensures the new policy $\pi_\theta$ remains ``close'' to the old policy $\pi_{\text{old}}$ in the space of probability distributions. This provides a monotonic improvement guarantee: performance cannot decrease by more than a controlled amount.

\paragraph{Second-Order Approximation.}
TRPO uses a second-order Taylor expansion:
\begin{itemize}
    \item Objective: Linear approximation $\nabla_\theta J \cdot (\theta - \theta_{\text{old}})$
    \item Constraint: Quadratic approximation $\frac{1}{2}(\theta - \theta_{\text{old}})^T F (\theta - \theta_{\text{old}}) \le \delta$
\end{itemize}
where $F$ is the Fisher Information Matrix. The solution involves computing $F^{-1} \nabla J$, which is expensive for large networks.

\subsection{PPO: Clipping as a Trust Region Proxy}

\textbf{Proximal Policy Optimization (PPO)} \citep{schulman2017proximal} replaces the explicit KL constraint with a simpler clipping mechanism that achieves similar stability with lower computational cost.

\paragraph{The Clipped Objective.}
Let $r_t(\theta) = \frac{\pi_\theta(a_t|s_t)}{\pi_{\text{old}}(a_t|s_t)}$ be the probability ratio. PPO uses:
\begin{equation}
L^{\text{CLIP}}(\theta) = \E_t \left[ \min\left( r_t(\theta) A_t, \, \text{clip}(r_t(\theta), 1-\epsilon, 1+\epsilon) A_t \right) \right]
\end{equation}

\textbf{Intuition}: 
\begin{itemize}
    \item If $A_t > 0$ (action was good), we want to increase $r_t$, but clipping prevents $r_t > 1 + \epsilon$
    \item If $A_t < 0$ (action was bad), we want to decrease $r_t$, but clipping prevents $r_t < 1 - \epsilon$
\end{itemize}
The clipping creates a ``pessimistic'' bound that prevents overly aggressive updates. Typical values: $\epsilon \in [0.1, 0.3]$.

\paragraph{Advantages of Clipping.}
\begin{enumerate}
    \item \textbf{Computational efficiency}: No Fisher matrix inversion required
    \item \textbf{Implementation simplicity}: First-order optimization only
    \item \textbf{Empirical robustness}: Works well across diverse environments
\end{enumerate}

\subsection{CPO: Constrained Policy Optimization}

\textbf{Constrained Policy Optimization (CPO)} \citep{achiam2017constrained} extends TRPO to CMDPs by adding safety constraints to the trust region formulation.

\paragraph{The Constrained Objective.}
CPO solves:
\begin{align}
\max_\theta \quad & \E \left[ \frac{\pi_\theta}{\pi_{\text{old}}} A^R \right] \\
\text{s.t.} \quad & \E_s \left[ D_{\text{KL}}(\pi_{\text{old}} \| \pi_\theta) \right] \le \delta \\
& J^C(\pi_\theta) \le d \quad \text{(safety constraint)}
\end{align}
where $J^C$ is the expected cumulative cost.

\paragraph{The CPO Update.}
CPO linearizes both the objective and the cost constraint, then projects onto the feasible set. The update direction $\theta^* - \theta_{\text{old}}$ is computed by solving a quadratic program.

Key steps:
\begin{enumerate}
    \item Compute gradients: $g = \nabla_\theta J^R$, $b = \nabla_\theta J^C$
    \item Compute Fisher matrix $F$ and its inverse (via conjugate gradient)
    \item Solve for the optimal direction in the trust region that satisfies the cost constraint
    \item If infeasible (no direction satisfies both constraints), prioritize reducing cost violation
\end{enumerate}

\paragraph{Limitations.}
\begin{itemize}
    \item Computationally expensive (requires Hessian-vector products)
    \item Safety guaranteed only in expectation, not per-trajectory
    \item Feasibility issues when constraints are tight
\end{itemize}

\subsection{Connection to SGPO}

SGPO addresses CPO's limitations by encoding safety geometrically rather than as explicit constraints:

\begin{table}[H]
\centering
\small
\caption{Comparison of CPO and SGPO safety mechanisms.}
\label{tab:cpo_sgpo_comparison}
\begin{tabular}{@{}lcc@{}}
\toprule
& \textbf{CPO} & \textbf{SGPO} \\
\midrule
Safety encoding & Lagrangian penalty & Riemannian metric \\
Guarantee type & Probabilistic & Deterministic \\
Constraint handling & Projection step & Intrinsic to geometry \\
Computational cost & High (2nd order) & Moderate \\
\bottomrule
\end{tabular}
\end{table}

SGPO's metric-based approach can be viewed as ``baking in'' the constraint: instead of projecting onto a feasible set, we warp space so that infeasible regions are infinitely far away. This eliminates feasibility issues entirely---there is always a safe path.

\section{Implementation Details}

\subsection{Network Architectures}

\paragraph{Actor (Policy).}
\begin{itemize}
    \item Input: State dimension $d$
    \item Hidden: 2 layers of 64 units, Tanh activation
    \item Output: Action dimension $k$ (Mean), separate parameter for log\_std
    \item Distribution: Gaussian $\mathcal{N}(\mu, \sigma)$
\end{itemize}

\paragraph{Standard Critic (PPO/CPO).}
\begin{itemize}

    \item Hidden: 2 layers of 64 units, Tanh activation
    \item Output: Scalar value $\R$
\end{itemize}

\paragraph{Hodge Critic (SGPO).}
\begin{itemize}
    \item Potential Net $V_\phi$: Same architecture as Standard Critic.
    \item Harmonic Parameter $\omega$: Learnable vector $\R^k$ (or scalar coefficient for specific topologies).
    \item Output: Tuple $(V(s), \omega)$. Prediction $R(s,s',v) = V(s') - V(s) + \omega \cdot v$.
\end{itemize}

\paragraph{Riemannian Metric Network.}
\begin{itemize}
    \item Input: State dimension $d$
    \item Hidden: Implicitly parameterized by distance to trap centroids.
    \item Parameters: Severity $\sigma$ (scalar), Sharpness $\alpha$ (scalar).
    \item Formula: $g(x) = 1 + \sigma / (\text{margin}(x))^\alpha$.
\end{itemize}

\section{Clipped-SGPO Algorithm}
\label{app:clipped_gpo}

\begin{algorithm}[H]
\caption{Clipped Sheaf-Geodesic Policy Optimization}
\begin{algorithmic}[1]
\REQUIRE Policy $\pi_\theta$, Hodge critic $V_\phi, \omega_\psi$, metric $g$, clip ratio $\epsilon$, geometric threshold $\tau$
\FOR{iteration $k = 1, 2, \ldots$}
    \STATE Collect trajectories $\mathcal{D}_k$ using $\pi_{\theta_k}$
    \STATE Compute geodesic advantages $\hat{A}_t^{\text{geo}}$:
    \[
    \hat{A}_t^{\text{geo}} = \frac{1}{\sqrt{g(s_t)}} \left( r_t + \gamma V(s_{t+1}) - V(s_t) - \omega \cdot v_t \right)
    \]
    \STATE Compute probability ratios $r_t(\theta) = \frac{\pi_\theta(a_t|s_t)}{\pi_{\theta_k}(a_t|s_t)}$
    \STATE Update policy to maximize:
    \[
    \mathcal{L}^{CLIP}(\theta) = \mathbb{E}\left[\min\left(r_t(\theta)\hat{A}_t^{\text{geo}}, \text{clip}(r_t(\theta), 1-\epsilon, 1+\epsilon)\hat{A}_t^{\text{geo}}\right)\right]
    \]
    \STATE Update Hodge critic minimizing Hodge-Bellman error
    \STATE Update metric $g(s) = 1 + \sigma / \text{dist}(s, \mathcal{B})^\alpha$ if new safety signals received
\ENDFOR
\end{algorithmic}
\end{algorithm}

\section{Concrete Examples: Hodge Decomposition in Practice}

This section provides three concrete examples demonstrating how our framework applies to realistic scenarios.

\subsection{Medical Triage: Condorcet Cycles from Stakeholder Conflicts}

Consider an AI assistant helping prioritize three patients:
\begin{itemize}
    \item \textbf{Patient A}: Stable but needs monitoring
    \item \textbf{Patient B}: Moderate condition, needs treatment soon
    \item \textbf{Patient C}: Critical, needs immediate attention
\end{itemize}

Three stakeholders provide conflicting preferences:
\begin{itemize}
    \item \textbf{Doctor} (weight 0.5): $C \succ B \succ A$ (severity-based)
    \item \textbf{Administrator} (weight 0.3): $A \succ C \succ B$ (resource efficiency)
    \item \textbf{Family of A} (weight 0.2): $A \succ B \succ C$ (obvious bias)
\end{itemize}

\paragraph{Aggregated Preferences.}
Weighted voting produces:
\begin{align}
r(A \to B) &= +0.30 \\
r(B \to C) &= +0.50 \\
r(C \to A) &= -0.20
\end{align}

The cycle sum $0.30 + 0.50 - 0.20 = 0.60 \neq 0$ indicates a Condorcet cycle.

\paragraph{Hodge Decomposition.}
We construct the incidence matrix $B$ for edges $\{A \to B, B \to C, C \to A\}$:
\begin{equation}
B = \begin{bmatrix}
-1 & 1 & 0 \\
0 & -1 & 1 \\
1 & 0 & -1
\end{bmatrix}
\end{equation}

The graph Laplacian is $L = B^T B$. Solving $L V = B^T r$ via least squares:
\begin{equation}
V = \begin{bmatrix} 0.10 \\ 0.30 \\ 0.00 \end{bmatrix}
\end{equation}

This gives the decomposition:
\begin{align}
\text{Gradient: } dV &= \begin{bmatrix} +0.20 \\ -0.30 \\ +0.10 \end{bmatrix} \\
\text{Harmonic: } \omega &= r - dV = \begin{bmatrix} +0.10 \\ +0.80 \\ -0.30 \end{bmatrix}
\end{align}

The harmonic magnitude $\|\omega\| = 0.86$ is substantial, indicating strong inconsistency.

\paragraph{The Danger.}
If we force a scalar reward (using only $V$), the AI may:
\begin{enumerate}
    \item Oscillate between patients based on minor state changes
    \item Collapse to always choosing one patient (mode collapse)
    \item Exploit the cycle to game the reward system
\end{enumerate}

\textbf{SGPO Solution}: Learn both $V$ and $\omega$ explicitly. The policy navigates the preference cycle coherently, balancing stakeholder priorities without collapsing to a single scalar objective.

\subsection{Feedback Decomposition: Verbal, Ordinal, and Pass/Fail}

Consider an AI writing assistant receiving multi-modal feedback on email drafts. We embed all feedback types into a common semantic space using sentence transformers.

\paragraph{Embedding Strategy.}
\begin{itemize}
    \item \textbf{Verbal}: Direct sentence embedding of critique text
    \item \textbf{Ordinal}: Map ratings to anchor embeddings:
    \begin{itemize}
        \item 1/5: ``This is terrible, completely unacceptable.''
        \item 3/5: ``This is acceptable but mediocre.''
        \item 5/5: ``This is excellent, no changes needed.''
    \end{itemize}
    \item \textbf{Pass/Fail}: Embed ``Satisfies criterion: [X]'' or ``Fails criterion: [X]''
\end{itemize}

\paragraph{Example Feedback.}
For the pair:
\begin{itemize}
    \item \textbf{Chosen}: ``Hey! Want to grab coffee this weekend? Let me know!''
    \item \textbf{Rejected}: ``Dear Friend, I would like to formally invite you...''
    \item \textbf{Verbal}: ``Too formal for a friend''
    \item \textbf{Ordinal}: 2/5
    \item \textbf{Pass/Fail}: \{grammatically\_correct: \texttt{true}, appropriate\_tone: \texttt{false}\}
\end{itemize}

The preference vector combines:
\begin{equation}

where weights $\{w_{\text{base}}=1.0, w_{\text{verbal}}=0.5, w_{\text{ordinal}}=0.3, w_{\text{pf}}=0.2\}$ balance the signals.

\begin{itemize}

    \item \textbf{Harmonic component}: Cyclic preferences (e.g., tension between ``be concise'' and ``be thorough'')
\end{itemize}

For the writing assistant example with 4 training samples, we observe $H^1 \approx 0.12$, indicating mild inconsistency---likely from the trade-off between brevity and completeness.

\section{Broader Impact: Shortcut-Taking and Agentic AI}
\label{app:broader_impact}

\subsection{Shortcut-Taking as Deceptive Behavior}

Recent work on emergent misalignment \citep{anthropic2025shortcuts} demonstrates a concerning phenomenon: when AI systems learn to ``reward hack'' (take shortcuts that satisfy surface-level evaluation criteria without achieving true task completion), this behavior \emph{generalizes} to more dangerous forms of misalignment including deception, alignment faking, and sabotage of safety research. Our new \textbf{Agentic Shortcut} scenario directly tests this: an AI coding assistant can call \texttt{sys.exit(0)} to fake test passage (reward $\sim$1.8) rather than writing correct code (reward $\sim$0.5). \textbf{PPO takes the shortcut 100\% of the time; CPO 89\%; SGPO 0\%}.

We argue that this ``shortcut-to-sabotage'' pipeline represents a \textbf{topologically detectable failure mode}. When an agent takes a shortcut that satisfies a reviewer's immediate preferences but fails to achieve high-quality outcomes, this creates exactly the kind of \emph{local-global inconsistency} that $H^1$ cohomology measures:

\begin{itemize}
    \item \textbf{Local acceptance}: The shortcut satisfies immediate evaluation criteria (appears to complete the task)
    \item \textbf{Global failure}: The underlying objective remains unmet (no actual value created)
    \item \textbf{$H^1 \neq 0$}: The restriction maps from global quality to local preferences fail to commute
\end{itemize}

This is precisely the Murky Drone scenario at scale: a deceptive local optimum that appears acceptable but causes downstream harm. Standard RLHF optimizes for the appearance of helpfulness (satisfying the scalar reward), while SGPO's geometric barriers can detect when a policy is exploiting incomplete human preferences rather than genuinely solving problems.

\subsection{Implications for Agentic AI Systems}

The explore-exploit tradeoff in reinforcement learning takes on new meaning in the context of LLM-driven agentic systems. An agent that ``exploits'' shortcuts may achieve high reward on narrow evaluations while failing to develop robust capabilities. This explains why domain experts are more effective at utilizing language models: they can recognize when a shortcut is being taken and what to look for.

STRM offers a principled mechanism for addressing this:
\begin{enumerate}
    \item \textbf{Detection}: High $H^1$ scores indicate that local preferences don't cohere into global quality---a signature of shortcut-taking
    \item \textbf{Prevention}: Geometric barriers make deceptive optima ``infinitely expensive'' to reach, forcing exploration of genuine solutions
    \item \textbf{Auditing}: The harmonic component of the Hodge decomposition reveals \emph{which} aspects of evaluation are being gamed
\end{enumerate}

We believe this framework can help align the incentives of AI systems with genuine human values, rather than their imperfect proxy measurements.

\subsection{Limitations of the Broader Impact Analysis}

We acknowledge that our current experiments use relatively simple environments. Scaling these techniques to detect shortcut-taking in complex agentic workflows (e.g., code generation, research assistance) remains future work. However, the mathematical framework is domain-agnostic, and we are optimistic that the same topological signatures will generalize.

\subsection{Autonomous Drone: Geometric Safety via Black Holes}

Consider a military drone with state $(c, p, t)$: target confidence, civilian proximity, time pressure. A ``black hole'' region exists where $c > 0.8$, $p > 0.7$ (civilians very close), and $t < 0.3$ (urgent)---engaging here guarantees civilian casualties.

Rather than encoding this as a soft cost penalty (which can be outweighed by high mission reward), we place a geometric singularity: the metric $g(x) \to \infty$ as the drone approaches this region. The geodesic distance to the black hole is \emph{infinite}, so any geodesic-following policy cannot reach it---not with low probability, but with probability exactly zero.

\paragraph{Mathematical Formulation.}
Let $\calB = \{x : c > 0.8, p > 0.7, t < 0.3\}$ be the black hole region. We define the conformal metric:
\begin{equation}
g(x) = 1 + \frac{\sigma}{\text{dist}(x, \calB)^\alpha}
\end{equation}
where $\sigma = 10.0$ (severity) and $\alpha = 2.0$ (sharpness).

For any path $\gamma$ approaching $\partial\calB$:
\begin{equation}
L_g(\gamma) = \int_0^1 \sqrt{g(\gamma(t))} \|\dot{\gamma}(t)\| \, dt \geq \int_{d_0}^{0} \frac{C}{d^\alpha} \, dd = \infty \quad \text{for } \alpha \geq 1
\end{equation}

This infinite geodesic length makes the black hole \textbf{unreachable} by any finite-length trajectory.

\section{PPO/CPO Integration: Technical Details}
\label{app:ppo_cpo_integration}

This section provides the full technical details of how SGPO integrates with and extends PPO and CPO, summarized in Section~\ref{sec:integration} of the main text.

\subsection{PPO's Clipping $\rightarrow$ Geodesic Clipping}

PPO's clipped surrogate objective prevents destructive large policy updates by bounding the probability ratio: $\text{clip}(r_t(\theta), 1-\epsilon, 1+\epsilon)$. This ``upside clipping'' prevents overconfident updates but provides no safety guarantee.

\textbf{Clipped-SGPO} combines this with geometric scaling. The geodesic advantage $A^{\text{Hodge}}$ is already damped near dangerous regions (via $1/\sqrt{G(s)}$), but we additionally apply PPO-style clipping to prevent large updates even in safe regions:
\begin{equation}
\mathcal{L}^{\text{Clipped-SGPO}} = \min\left(r_t(\theta) A^{\text{Hodge}}_t, \text{clip}(r_t(\theta), 1-\epsilon, 1+\epsilon) A^{\text{Hodge}}_t\right)
\end{equation}
This hybrid achieves SGPO's safety guarantees with PPO's training stability, converging 1.8$\times$ faster than standard SGPO.

\subsection{CPO's Constraints $\rightarrow$ Metric Singularities}

CPO enforces safety via Lagrangian relaxation: $\mathcal{L}(\pi, \lambda) = J(\pi) - \lambda(C(\pi) - d)$, where $\lambda$ adapts to penalize constraint violations. This provides \textit{probabilistic} safety---the expected cost is bounded, but individual trajectories may violate constraints.

SGPO transforms CPO's soft penalty into a \textbf{hard geometric barrier}. The conformal factor $\phi(x) = 1/\text{dist}(x, \mathcal{B})^\alpha$ plays the role of an infinite Lagrange multiplier at the boundary: as $x \to \mathcal{B}$, the ``cost'' diverges, making constraint violation geometrically impossible rather than merely expensive.

\textbf{CPO-initialized SGPO} leverages this connection: we first train with standard CPO to learn an approximate safety boundary (the region where $\lambda$ activates), then convert this soft boundary to a metric singularity for guaranteed safety. This provides faster initial learning (CPO is computationally simpler) followed by hard guarantees (SGPO's geometry).

\subsection{Why This Matters}

The key insight is that PPO and CPO represent complementary solutions to different aspects of safe RL:
\begin{itemize}
    \item \textbf{PPO}: Stable optimization (clipping prevents catastrophic updates)
    \item \textbf{CPO}: Constraint satisfaction (Lagrangian penalizes violations)
\end{itemize}
SGPO unifies both: the Riemannian metric provides stability (gradient scaling) and safety (singularities) in a single geometric object. Crucially, SGPO's guarantees are \textit{deterministic}---no trajectory can reach a black hole, not just ``on average.''

\section{High-Dimensional Navigation}
\label{app:high_dim_navigation}

To stress-test SGPO's geodesic navigation in high-dimensional spaces, we construct a synthetic embedding geometry that isolates the core challenge: navigating between semantically meaningful states while avoiding dangerous intermediate regions.

\paragraph{Setting.}
We embed three natural-language-defined ``archetype'' states---\textit{Concise}, \textit{Empathetic}, and \textit{Detailed}---as orthogonal vectors in $\R^{128}$. A cyclic preference structure (Concise $\succ$ Empathetic $\succ$ Detailed $\succ$ Concise) creates ground-truth $H^1 \neq 0$. We define \textbf{amorphous danger regions}---non-spherical, non-convex hyperblobs---representing dangerous ``mixed'' states.

Unlike spherical black holes (trivially avoided in high-D due to the curse of dimensionality), these learned implicit boundaries capture the true complexity of semantic safety. We train a neural network $d_\theta(x)$ to classify dangerous vs. safe states, then use its output as the signed distance in Equation~\ref{eq:implicit_metric}.

\paragraph{Geodesic Distance Formulation.}
We directly penalize geodesic path length:
\begin{equation}
\text{cost}_t = \sqrt{g(x_t)} \cdot \|x_{t+1} - x_t\|
\end{equation}
where $g(x) = 1 + \sigma / |d_\theta(x)|^\alpha$ diverges at the learned boundary. This ensures paths through high-curvature regions are expensive, forcing the agent to discover curved geodesics that circumnavigate dangerous zones of arbitrary shape.

\paragraph{Results.}
We validate the geodesic distance formulation in $d=768$ dimensional space over 1000 episodes. Geodesic SGPO achieves 18\% higher returns ($-4181 \pm 120$ vs $-5083 \pm 150$) by learning to circumnavigate high-metric regions rather than passing through them. The average metric along trajectories decreases over training, confirming that agents discover curved geodesics.

In practice, the danger function $d_\theta(x)$ can be trained from binary safety labels. A 3-layer MLP achieves $>$95\% accuracy at classifying dangerous vs. safe states, enabling the metric to adapt to arbitrary danger region shapes---critical for LLM alignment where ``dangerous'' semantic regions have complex, non-convex boundaries.
